
```kotlin
class AnalyzeRule {
    
    fun getString(rule: String): String {
        return when {
            rule.startsWith("@css:") -> parseByCss(rule)
            rule.startsWith("@xpath:") -> parseByXPath(rule)
            rule.startsWith("$.") -> parseByJsonPath(rule)
            rule.startsWith(":") -> parseByRegex(rule)
            else -> parseByDefault(rule)
        }
    }
}
```
